537 public void keyPressed(KeyEvent e) {
538 super.keyPressed(e);

540 if (e.getKeyCode() == 32 && toolStatus==ToolStatus.TOOL_LINK_CSMA) {
541 // render the CSMA link...
542 painter.addAndPrintLink(new CsmaLinkPainter(nodesInCsma,painter.getNodes()));
543 // instantiate connection dialog...

554
555 // record each mouse click on canvas...
556 // do accordingly...
557 this.painter.addMouseListener(new MouseAdapter() {
558 @Override
559 public void mouseClicked(MouseEvent e) {
560 super.mouseClicked(e);
561 DebuggingHelper.Debugln("Clicked at x : "+e.getX()+" y : "+e.getY());
562 switch (toolStatus) {
563 // if tool is 'node' tool then...
564 case TOOL_NODE : {
565 // create a new node at given point...
566 NodePainter node = new NodePainter(e.getX()-10, e.getY()-10, 20,String.valueOf(painter.getNodes().size()));
567 // add that node to painter for painting on canvas...
568 painter.addAndPrintNode(node);
569 } break;
570
571 // if tool is 'link' tool then...
572 case TOOL_LINK : {
573 int collision = -1;
574 boolean successfulClick = false;
575 // for first successful click....
576 // check if collision with any node...
577 collision = painter.pointCollideWithAny(e.getX(), e.getY());
578 DebuggingHelper.Debugln("Collision : "+collision+" Clicks : "+clicks);
579 if (collision >= 0 && clicks%2!=0) {
580 // if collision then change the information label and increment the no. of clicks...
581 firstNode = collision;
582 lbl_info.setText("Connection Tool Selected: First node selected now, select second node.");
583 successfulClick = true;
584 }
585
586 // for second successful click...
587 // check if collision with any node...
588 if (collision >= 0 && clicks%2==0) {
589 // if collision then change the information label, increment the no. of clicks and open a device configuration dialog box...
590 lbl_info.setText("Connection Tool Selected: To create a link, click on two nodes sequentially.");
591 painter.addAndPrintLink(new P2pLinkPainter(painter.getNodes().get(firstNode), painter.getNodes().get(collision)));
592 successfulClick = true;
593 DebuggingHelper.Debugln(firstNode+" "+collision);
594 instantiateConnectionDialog();
595 ArrayList<Integer> nodes = new ArrayList<>();
596 nodes.add(firstNode);
597 nodes.add(collision);
598 dialogConnection.showDialog(dialogLink.getAllLinks(),dialogNetwork.getAllNetworks(),nodes, toolStatus);
599 firstNode = -1;
600 }
601 incrementClicks(successfulClick);
602 } break;
603
604 case TOOL_LINK_CSMA: {
605 int collision = -1;
606
607 collision = painter.pointCollideWithAny(e.getX(), e.getY());
608 DebuggingHelper.Debugln("Collision : "+collision);
609 if (collision >= 0 && !nodesInCsma.contains(collision)) {
610 nodesInCsma.add(collision);
611 painter.addAndPrintRefNode(new NodePainter(painter.getNodes().get(collision).xPos-5, painter.getNodes().get(collision).yPos-5,30,"",Color.GREEN));
612 }
613 } break;
614
615 }

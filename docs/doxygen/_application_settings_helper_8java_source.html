125 public static ArrayList<NetworkLink> getDefaultLinks() {

133 if (params[params.length-1].equalsIgnoreCase(NetworkLink.LABEL_CSMA)) {
134 links.add(new CSMA(links.size(), params[0], params[1], params[2], params[3], ((params[4].equalsIgnoreCase("Y"))?(true):(false)), true));
135 } else { // default or p2p case...

139 DebuggingHelper.Debugln("Yes! Universal Settings has Default Links!");
